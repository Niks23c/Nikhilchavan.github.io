WITH Ranking AS (
  SELECT ProductLine, Gender, COUNT(Quantity) AS Frequency,
         RANK() OVER (PARTITION BY Gender ORDER BY COUNT(Quantity) DESC) AS Rank
  FROM Amazon
  GROUP BY Gender, ProductLine
)
SELECT ProductLine, Gender, Frequency 
FROM Ranking
WHERE Rank = 1;
