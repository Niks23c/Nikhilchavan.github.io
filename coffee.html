üìà Key KPIs

  • Total Revenue ‚Äì SUM of sales per product
  • Quantity Sold ‚Äì Total quantity sold across days
  • Gross Profit ‚Äì Revenue minus cost
  • Top Product by Sales, Profit, Quantity
-- Total Revenue
SELECT SUM(Sales) AS Total_Revenue FROM coffee_shop;

-- Total Quantity
SELECT SUM(Quantity) AS Total_Quantity FROM coffee_shop;

-- Gross Profit
SELECT SUM(Profit) AS Total_Profit FROM coffee_shop;

-- Top Products
SELECT TOP 5 Product, SUM(Sales) AS Total_Sales
FROM coffee_shop
GROUP BY Product
ORDER BY Total_Sales DESC;
